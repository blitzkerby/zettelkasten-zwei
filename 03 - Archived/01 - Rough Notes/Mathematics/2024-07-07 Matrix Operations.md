### $$
\begin{align}
	& \begin{cases} 
		y_1 = ax_1 + bx_2 \\ \\
		y_2 = cx_1 + dx_2 
	\end{cases}

	\\ \\
	
	\Longrightarrow & \ 
		\begin{pmatrix} 
			y_1 \\ 
			y_2 
		\end{pmatrix} = 
		\begin{pmatrix} 
			ax_1 + bx_2 \\ 
			cx_1 + dx_2 
		\end{pmatrix}

	\\ \\

	\Longrightarrow & \ 
		\begin{pmatrix} 
			y_1 \\ 
			y_2 
		\end{pmatrix} 
		= 
		\begin{pmatrix} 
			a & b \\ 
			c & d 
		\end{pmatrix}
		\begin{pmatrix} 
			x_1 \\ 
			x_2 
		\end{pmatrix}
	
\end{align}
$$
